% Minimal example
\documentclass{article}

% Meta-data for the title!
\title{This is an example.}
\author{Zane!}
\date{\today}

\begin{document}
% How to make a title
\maketitle

% Typing sentences, using \vspace{}, and line breaks.
This is a sentence of my choice. Mary had a little lamb. The    quick brown fox jumped over the lazy dog. \\ \vspace{1in} %cm, px, em, etc.

% How to use \hspace{}.
New \hspace{7cm} line! \\

% How to make bold text.
\textbf{This sentence is very important.} \\

% How to make italicized text.
\textit{I want this sentence to be in italics.}

% How to underline.
\underline{The big black bug bled blue-black blood.}

% Center aligned.
\begin{center}
    But the other black bug bled blue.
\end{center}

\end{document}
